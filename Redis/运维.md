#### 日常运维

1.安全批量删除大量的KEY
```
redis-cli -h x.x.x.x -a 'passwd' -n 0 --scan --pattern 'sample_key_*' |xargs -L 500 redis-cli -h x.x.x.x -a 'passwd' -n 0 del
```

2.流量告警监控
```
# 一般流量告警由GETALL命令引发
redis-cli -h x.x.x.x -a 'passwd' monitor > monitor.log
```

3.流量告警监控-定期版
```
import subprocess
import re
from collections import defaultdict
import time
import argparse
import redis

class RedisMonitor:
    def __init__(self, host='localhost', port=6379, password=None):
        self.host = host
        self.port = port
        self.password = password
        self.key_pattern = re.compile(r'^\d+\.\d+ \[(\d+) .*?\] ".*?" "(.*?)"')
        self.redis_client = redis.Redis(
            host=host,
            port=port,
            password=password,
            decode_responses=False
        )

    def start_monitor(self, output_file=None):
        """启动redis-cli monitor子进程并记录到文件"""
        if not output_file:
            output_file = f"redis_monitor_{int(time.time())}.log"

        cmd = ['redis-cli', '-h', self.host, '-p', str(self.port)]
        if self.password:
            cmd.extend(['-a', self.password])
        cmd.append('monitor')

        try:
            with open(output_file, 'w') as f:
                process = subprocess.Popen(
                    cmd,
                    stdout=subprocess.PIPE,
                    stderr=subprocess.PIPE,
                    encoding="utf-8"
                )

                print(f"Monitoring started. Logging to {output_file}")
                print("Press Ctrl+C to stop...")

                try:
                    for line in process.stdout:
                        f.write(line)
                        f.flush()
                except KeyboardInterrupt:
                    process.terminate()
                    print("\nMonitoring stopped.")

        except Exception as e:
            print(f"Error: {e}")

    def analyze_log(self, log_file):
        """分析日志文件并统计key请求"""
        stats = []

        try:
            with open(log_file, 'r') as f:
                for line in f:
                    match = self.key_pattern.search(line)
                    if match:
                        dbnum = match.group(1)
                        key = match.group(2)
                        stats.append((dbnum, key))

            # 统计每个key在db中的请求数
            counter = defaultdict(int)
            for dbnum, key in stats:
                counter[(dbnum, key)] += 1

            # 按请求数降序排序
            sorted_stats = sorted(counter.items(), key=lambda x: x[1], reverse=True)

            print("\nKey Request Statistics with Memory Usage (Top 30):")
            print("------------------------------------------------")
            print(f"{'Count':<8} {'Size':<10} {'DB':<4} {'Key':<40}")
            print("-" * 70)

            # 只处理前20个key
            for (dbnum, key), count in sorted_stats[:30]:
                try:
                    # 选择正确的数据库
                    self.redis_client.execute_command('SELECT', dbnum)
                    # 获取key大小(字节)
                    size = self.redis_client.memory_usage(key) or 0
                    size_str = f"{size/1024:.2f}KB" if size >= 1024 else f"{size}B"
                except Exception as e:
                    size_str = "N/A"

                print(f"{count:<8} {size_str:<10} {dbnum:<4} {key:<40}")

            return sorted_stats

        except Exception as e:
            print(f"Error analyzing log: {e}")
            return None

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description='Redis Monitor Tool')
    parser.add_argument('--host', default='localhost', help='Redis host')
    parser.add_argument('--port', type=int, default=6379, help='Redis port')
    parser.add_argument('--password', help='Redis password')
    parser.add_argument('--log', help='Output log file path')
    parser.add_argument('--analyze', help='Analyze existing log file')

    args = parser.parse_args()

    monitor = RedisMonitor(host=args.host, port=args.port, password=args.password)

    if args.analyze:
        monitor.analyze_log(args.analyze)
    else:
        monitor.start_monitor(args.log)
```
